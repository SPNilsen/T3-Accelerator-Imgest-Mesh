% --- Headings: blue + each H1 starts a new page ---
% We avoid titlesec; use core \@startsection hooks.
\makeatletter

% H1 in markdown -> \section in LaTeX (Pandoc default)
% Force new page before every \section, and color it Trace3Blue
\let\OldSection\section

% Amount to push H1 down on its own page (tune if needed)
\newcommand{\HOneDownshift}{0.40\textheight}

\renewcommand\section{%
  \clearpage
  \vspace*{\HOneDownshift}%
  \@startsection{section}{1}{\z@}%
    {-3.5ex \@plus -1ex \@minus -.2ex}%
    {2.3ex \@plus .2ex}%
    {\normalfont\Large\bfseries\color{Trace3Blue}}%
}

\renewcommand\subsection{%
  \@startsection{subsection}{2}{\z@}%
    {-3.25ex\@plus -1ex \@minus -.2ex}%
    {1.5ex \@plus .2ex}%
    {\normalfont\large\bfseries\color{Trace3Blue}}%
}

\renewcommand\subsubsection{%
  \@startsection{subsubsection}{3}{\z@}%
    {-3.25ex\@plus -1ex \@minus -.2ex}%
    {1.5ex \@plus .2ex}%
    {\normalfont\normalsize\bfseries\color{Trace3Blue}}%
}

\makeatother

% --- Code blocks: EXACT colors + padding (no tcolorbox/fvextra) ---
% Pandoc emits Shaded + Highlighting. We'll implement both.
\usepackage{fancyvrb}

% This env is where the code characters are printed.
\DefineVerbatimEnvironment{Highlighting}{Verbatim}{
  commandchars=\\\{\},
  fontsize=\small,
  formatcom=\color{CodeText}\CodeFont
}

% Shaded wraps Highlighting. We'll mimic CSS-style background + padding.
\newenvironment{Shaded}{%
  \par\noindent
  \setlength{\fboxsep}{4pt}% padding like CSS
  \colorbox{CodeBG}{%
    \begin{minipage}{0.97\linewidth}%
      \vspace{2pt}%
      \color{CodeText}\CodeFont\small%
}{%
      \vspace{2pt}%
    \end{minipage}%
  }%
  \par
}
